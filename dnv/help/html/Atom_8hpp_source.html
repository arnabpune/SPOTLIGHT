393void Atom::evaluateHybridization()
394{
395 //if(!atom_type) {throw EmptyAtomTypeException();}
396 switch(atom_type[0])
397 {
398 case 'C': hybridization=valency-1; break; //4 valency -> sp[3]
399 case 'S': case 'O': hybridization=valency+1; break; //1 valency -> sp[2]
400 case 'P': case 'N': hybridization=(valency==4)?3:valency; break; //3 valency -> sp[3]
401 case 'F': case 'H': hybridization=0; break;
402 }
403}
