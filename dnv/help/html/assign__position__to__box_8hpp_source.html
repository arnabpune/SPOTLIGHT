27/*
28Assign a position to a cell in a 3D grid:
29In the previous function we defined our origin r_o, our box d-spacing d and the box vectors a,b,c.
30In this section we are now assuming Nx = a/d, Ny = b/d and Nz = c/d. This allows us to assign a
31unique box to a coordinate in such a PERIODIC box, by using the formula
32{i,j,k} = round(r_i-r_o_i)/d,round(r_j-r_o_j)/d,round(r_k-r_o_k)/d with correction for periodic conditions.
33This formula implies that the rounded off distance divided by the d-spacing is equivalent to an index,
34or is atleast the nearest index.
35The other method would be to search for the nearest point but this is much more complex in timescale
36*/
37
38template<int Nx,int Ny,int Nz>
39Eigen::Vector3i position2cell(Eigen::Vector3d r, Eigen::Vector3d r_o, double d){
40 //first we find the difference
41 Eigen::Vector3d rdiff; rdiff = r-r_o;
42 //then we use the predefined pbc correction to measure and correct the calculated positions
43 Eigen::Vector3i ri; ri = pbc_index_correction<Nx,Ny,Nz>( ((int)(round((rdiff(0))/d))),0,0 );
44 Eigen::Vector3i rj; rj = pbc_index_correction<Nx,Ny,Nz>( 0,((int)(round((rdiff(1))/d))),0 );
45 Eigen::Vector3i rk; rk = pbc_index_correction<Nx,Ny,Nz>( 0,0,((int)(round((rdiff(2))/d))) );
46 //we then extract the values to a meaningful box index and return.
47 Eigen::Vector3i boxindx( ri(0), rj(1), rk(2) );
