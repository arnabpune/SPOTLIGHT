299 sol->genpurpflag=false; overlap++;
300 ent+=(float)(-WATENT*((float)obp/3));

302 }
303 }
304 }
305 for(Atom* tA : m->getAtoms())
306 {
307 int nb=0;

309 for(int si=0;si<solvent.size();si++)
310 {
311 Atom* sol=solvent[si];
312 if(!(sol->genpurpflag)) continue;
313 if(sol->distanceFrom(tA)<rad) {nb++; if(bps[si]<3) bps[si]++;}
314 }
315 if(!nb) continue;
316 inter+=nb;
317 int lim=0;
318 switch(tA->getStandardValency())
319 {
320 case 1:
321 lim=5;
322 break;
323 case 2:
324 lim=4;
325 break;
326 case 3:
327 lim=2;
328 break;
329 case 4:
330 lim=1;
331 break;
332 }
333 float frac=(float)nb/lim;
334 if(frac>1) frac=1;
335 ret+=tA->getSolvationCorrection()*frac;
336 }
337 cout << overlap<<" overlapping solvent molecules omitted\n";
338 cout << "They produce a total entropy difference of: "<<ent<<" kJ/mol\n";
339 for(int si=0;si<solvent.size();si++)
340 {
341 if(!(solvent[si]->genpurpflag)) continue;
342 if(bps[si]!=solvent[si]->bindpoints) ent+=WATENT*(((float)(bps[si]-solvent[si]->bindpoints))/3);

350 std::pair<double,double> stats_mean(const std::vector<double>& v,const std::vector<double>& w=std::vector<double>())
351 {
352 if(!v.size()) return make_pair(0,0);
353 double res=0,res2;
354 double den=0;
355 int K=0;
356 for(const double& d : v)
357 {
358 if(w.size()) {res+=d*w[K]; res2+=d*d*w[K]; den+=w[K++];}
359 else {res+=d; res2+=d*d*w[K]; den++;}
360 }
361 double mean=res/den;
362 double stdev=::sqrt(res2/den-mean*mean);
363 return make_pair(mean,stdev);
